searchProduct =>

SELECT P.PRODUCT_ID, P.NAME, P.PRICE, NVL(O.PERCENTAGE_DISCOUNT,0),
ROUND(AVG_RATING(P.PRODUCT_ID),1)
FROM PRODUCT P LEFT JOIN OFFER O 
ON(O.PRODUCT_ID = P.PRODUCT_ID AND O.END_DATE>SYSDATE)
WHERE LOWER(P.NAME) LIKE %s;

returnOffers =>

SELECT PRODUCT.NAME, PRODUCT.DESCRIPTION, PRODUCT.PRICE, PRODUCT.PRODUCT_ID, 
OFFER.PERCENTAGE_DISCOUNT
FROM OFFER LEFT OUTER JOIN PRODUCT
ON(OFFER.PRODUCT_ID = PRODUCT.PRODUCT_ID)
WHERE OFFER.END_DATE > SYSDATE;
