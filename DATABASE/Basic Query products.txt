categoryList =>

SELECT CATEGORY_ID, CATEGORY_NAME FROM CATEGORY;

returnAddProduct =>

SELECT CATEGORY_NAME FROM CATEGORY;

SELECT SELLER_ID FROM SELLER WHERE EMAIL_ID=:email_id;

SELECT COUNT(*) FROM PRODUCT WHERE LOWER(NAME) = :pname AND SELLER_ID = :sel_id;

returnProductCat =>

SELECT CATEGORY_NAME FROM CATEGORY WHERE CATEGORY_ID = :cat_id;

SELECT NAME, DESCRIPTION, PRICE, PRODUCT_ID FROM PRODUCT WHERE CATEGORY_ID= :cat_id;

returnProductDetails=>

SELECT CUSTOMER_ID FROM CUSTOMER WHERE EMAIL_ID = :cus_email;

returnAddOffer =>

SELECT SELLER_ID FROM SELLER WHERE EMAIL_ID =: email_id;

SELECT PRODUCT_ID FROM PRODUCT 
WHERE SELLER_ID =: id AND NAME =: prod_name;

SELECT MAX(OFFER_NUMBER)
FROM OFFER
GROUP BY PRODUCT_ID, SELLER_ID
HAVING PRODUCT_ID =: prod_id AND SELLER_ID =: id;

returnCheckOut =>

SELECT FIRST_NAME||' '||LAST_NAME AS NAME, CUSTOMER_ID
FROM CUSTOMER
WHERE EMAIL_ID = :cus_email;
